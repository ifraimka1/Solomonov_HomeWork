//Задание 3. Вводится число от 0 до 5. Если число не попадает в данный диапазон, программа требует ввести число заново.
	//На основании введенного числа ввести N чисел и вывести на экран сумму введенных N чисел.

#include <iostream>
using namespace std;

int main() {
	setlocale(LC_ALL, "Rus");

	int N, t, sum = 0;
	do {
		cout << "Введите число (количество вводимых чисел)." << endl;
		cin >> N;
	} while (!((N >= 0) && (N <= 5)));

	cout << "Введите " << N << " чисел." << endl;
	for (int i = 0; i < N; i++) {
		cin >> t;
		sum += t;
	}
	cout << "Сумма введенных чисел равна " << sum << endl;
}
