//Задание 5. Вводятся 5 чисел: x, y, z (габариты кирпича), u, v (размеры отверстия). 

#include <iostream>
using namespace std;

int main() {
	setlocale(LC_ALL, "Rus");

	int x, y, z;
	int u, v;
	while (true) {
		cout << "Введите размеры кирпича (x, y, z)." << endl;
		cin >> x >> y >> z;
		cout << "Введите размеры отверстия (u, v)." << endl;
		cin >> u >> v;

		if ((x <= u && ((y <= v) || (z <= v))) || (y <= u && ((x <= v) || (z <= v))) || (z <= u && ((x <= v) || (y <= v)))) {
			cout << "Кирпич пролезет в отверстие." << endl;
		}
		else {
			cout << "Кирпич не пролезет в отверстие." << endl;
		}

		cout << "Введите 0, чтобы завершить программу." << endl << "Введите любое другое число, чтобы перезапустить программу." << endl;
		cin >> x;
		if (x == 0) { break; }
	}
	cout << "Программа завершена." << endl;
}
